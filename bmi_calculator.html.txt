<script>
document.getElementById('bmiForm').addEventListener('submit', function(event) {
// Prevent the default form submission behavior (which causes a page refresh)
event.preventDefault();

// 1. Get Input Values
const weightKg = parseFloat(document.getElementById('weight').value);
const heightCm = parseFloat(document.getElementById('height').value);

// Basic validation
if (isNaN(weightKg) || isNaN(heightCm) || weightKg <= 0 || heightCm <= 0) {
document.getElementById('bmi-value').textContent = 'Error';
document.getElementById('bmi-category').textContent = 'Please enter valid numbers.';
document.getElementById('result-area').classList.remove('hidden');
return;
}

// 2. Calculation
// Convert height from cm to meters (m)
const heightM = heightCm / 100;

// Formula: BMI = weight (kg) / [height (m)]Â²
const bmi = weightKg / (heightM * heightM);
const roundedBmi = bmi.toFixed(1); // Round to one decimal place

// 3. Determine Category
let category = '';
let colorClass = '';

if (bmi < 18.5) {
category = 'Underweight';
colorClass = 'text-yellow-600';
} else if (bmi >= 18.5 && bmi < 25) {
category = 'Normal weight';
colorClass = 'text-green-600';
} else if (bmi >= 25 && bmi < 30) {
category = 'Overweight';
colorClass = 'text-orange-600';
} else {
category = 'Obesity';
colorClass = 'text-red-600';
}

// 4. Display Results
const resultArea = document.getElementById('result-area');
const bmiValueDisplay = document.getElementById('bmi-value');
const bmiCategoryDisplay = document.getElementById('bmi-category');

// Update Text Content
bmiValueDisplay.textContent = roundedBmi;
bmiCategoryDisplay.textContent = category;

// Update Color based on Category
// Reset existing color classes and apply new one
bmiCategoryDisplay.className = `text-lg font-medium mt-2 ${colorClass}`;

// Show the result area
resultArea.classList.remove('hidden');

// Scroll to the result area on small screens
resultArea.scrollIntoView({ behavior: 'smooth' });
});

// ðŸŸ¢ Register the Service Worker (Enables offline use and installability)
if ('serviceWorker' in navigator) {
window.addEventListener('load', () => {
// Ensure the path to the service worker is correct relative to the root of your GitHub repository
navigator.serviceWorker.register('/service-worker.js')
.then(reg => console.log('Service Worker registered: ', reg))
.catch(err => console.log('Service Worker registration failed: ', err));
});
}
